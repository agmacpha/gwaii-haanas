ints <- st_intersection(ghpc, h) %>%
  mutate(intersect_area = st_area(.)) %>%
  dplyr::select(HABITAT, location_name, intersect_area) %>%
  st_drop_geometry() %>%
  as_tibble() %>%
  mutate(ia = as.integer(intersect_area),
         prop = round((ia / 70685.83),5)) %>%
  inner_join(., g2, by = c("location_name" = "location_name")) %>%
  select(location_name, HABITAT, prop, species_code, individual, abundance_type, recording_date) %>%
  mutate(individual = case_when(
             abundance_type == "TMTC" ~ 4, #Conditional based on spp richness?
             abundance_type == "CI 1" ~ 1,
             abundance_type == "CI 2" ~ 5,
             abundance_type == "CI 3" ~ 20,
             TRUE ~ as.numeric(individual)))
## Warning: attribute variables are assumed to be spatially constant throughout all geometries
i2 <- ints %>%
    dplyr::filter(!species_code %in% c("BOCH","AMCR","BCCH")) %>%
    select(location_name, HABITAT, prop, species_code, individual, recording_date) %>%
    distinct() %>%
    mutate(year = lubridate::year(recording_date)) %>%
    group_by(location_name, year, HABITAT, prop, species_code) %>%
    mutate(individual = max(individual)) %>%
    ungroup() %>%
    group_by(location_name, year, species_code) %>%
    add_count(name = "rich") %>%
    ungroup() %>%
    distinct() %>%
    pivot_wider(names_from = species_code, values_from = individual, values_fill = 0) %>%
    as_tibble() %>%
    mutate_if(is.integer, as.numeric)
  multi_type <- i2 %>%
    select(location_name, HABITAT, prop, recording_date, rich, year) %>%
    distinct()
  t3 <- vegan::rda(i2[,-c(1:6)] ~ HABITAT + Condition(rich,prop), data = multi_type)
  vegan::ordistep(t3)
